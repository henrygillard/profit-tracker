Output: calculateOrderProfit returns feeSource; upsertOrder writes feeSource at write time; syncPayouts sets 'verified'; refund handler preserves feeSource.

<!-- feeSource state machine rules (from research) -->
```
'pending'   → paymentGateway.includes('shopify_payments') AND shopifyPaymentsFee === 0
'verified'  → set ONLY by syncPayouts after confirmed settled payout (never by upsertOrder alone)
'estimated' → third-party gateway; OR explicit passthrough from existingFeeSource param
```

<!-- Anti-patterns (from research) -->
- Do NOT set feesTotal = null for pending orders — feesTotal stays 0, feeSource = 'pending'
- Do NOT add a 4th status value
- Do NOT re-derive feeSource from fee amount alone in refund handler — read order.profit.feeSource
- upsertOrder must NOT set feeSource = 'verified' — only syncPayouts does that

```javascript
/**
 * Determine feeSource from payment gateway and fee amount.
 * Only syncPayouts may set 'verified' — this function never returns 'verified'.
 * @param {string} paymentGateway
 * @param {number} shopifyPaymentsFee
 * @returns {'pending'|'estimated'}
 */
function determineFeeSourceFromOrder(paymentGateway, shopifyPaymentsFee) {
  if (paymentGateway && paymentGateway.includes('shopify_payments')) {
    return (parseFloat(shopifyPaymentsFee) || 0) > 0 ? 'estimated' : 'pending';
  }
  return 'estimated';
}
```

**Strategy: spy on upsertOrder to capture what feeSource value it receives.**

Because `prisma.$transaction` is mocked (short-circuits before actual Prisma calls execute), asserting on the Prisma mock alone cannot detect if `existingFeeSource` is silently dropped in the webhook handler. The correct approach is to mock `upsertOrder` at the module level so the test can assert the exact 6th argument.

Add a new describe block at the end of `tests/webhooks.test.js`:

```javascript
const syncOrders = require('../lib/syncOrders');

describe('refunds/create preserves feeSource (FEEX-04)', () => {
  let upsertOrderSpy;

  beforeEach(() => {
    // Spy on upsertOrder so we can assert what existingFeeSource value it receives.
    // $transaction is mocked — we cannot assert Prisma call data directly from webhook tests.
    upsertOrderSpy = jest.spyOn(syncOrders, 'upsertOrder').mockResolvedValue(undefined);
  });

  afterEach(() => {
    upsertOrderSpy.mockRestore();
  });

  function makeRefundBody(orderId) {
    return JSON.stringify({
      order_id: orderId,
      transactions: [{ amount: '10.00' }],
    });
  }

  test('preserves feeSource: verified for a previously-verified order', async () => {
    const orderId = 99001;
    const body = makeRefundBody(orderId);
    const hmac = computeHmac(body);

    // Order exists in DB with feeSource: 'verified'
    prisma.order.findUnique
      .mockResolvedValueOnce({ id: `gid://shopify/Order/${orderId}`, shop: 'test.myshopify.com' })
      .mockResolvedValueOnce({
        id: `gid://shopify/Order/${orderId}`,
        shop: 'test.myshopify.com',
        totalPrice: '100.00',
        paymentGateway: 'shopify_payments',
        lineItems: [],
        profit: { feesTotal: 3.50, feeSource: 'verified' },
      });

    prisma.shopConfig.findFirst.mockResolvedValueOnce({ shopifyPlan: 'Basic' });
    prisma.shopConfig.upsert.mockResolvedValueOnce({});

    const res = await request(app)
      .post('/webhooks/refunds/create')
      .set('Content-Type', 'application/json')
      .set('X-Shopify-Hmac-Sha256', hmac)
      .send(body);

    expect(res.status).toBe(200);

    // Wait for async handler to complete
    await new Promise(resolve => setImmediate(resolve));

    expect(upsertOrderSpy).toHaveBeenCalled();
    // 6th argument is existingFeeSource — must be 'verified', not null or 'pending'
    const callArgs = upsertOrderSpy.mock.calls[0];
    expect(callArgs[5]).toBe('verified');
  });

  test('does not downgrade feeSource from verified to pending on refund', async () => {
    const orderId = 99002;
    const body = makeRefundBody(orderId);
    const hmac = computeHmac(body);

    prisma.order.findUnique
      .mockResolvedValueOnce({ id: `gid://shopify/Order/${orderId}`, shop: 'test.myshopify.com' })
      .mockResolvedValueOnce({
        id: `gid://shopify/Order/${orderId}`,
        shop: 'test.myshopify.com',
        totalPrice: '100.00',
        paymentGateway: 'shopify_payments',
        lineItems: [],
        profit: { feesTotal: 3.50, feeSource: 'verified' },
      });

    prisma.shopConfig.findFirst.mockResolvedValueOnce({ shopifyPlan: 'Basic' });
    prisma.shopConfig.upsert.mockResolvedValueOnce({});

    await request(app)
      .post('/webhooks/refunds/create')
      .set('Content-Type', 'application/json')
      .set('X-Shopify-Hmac-Sha256', hmac)
      .send(body);

    await new Promise(resolve => setImmediate(resolve));

    expect(upsertOrderSpy).toHaveBeenCalled();
    // feeSource must not be downgraded — must remain 'verified', not 'pending' or null
    const callArgs = upsertOrderSpy.mock.calls[0];
    expect(callArgs[5]).toBe('verified');
    expect(callArgs[5]).not.toBe('pending');
    expect(callArgs[5]).not.toBeNull();
  });
});
```
